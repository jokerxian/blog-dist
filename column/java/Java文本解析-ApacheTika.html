Apache Tika 实现文档内容解析 ​

1，导入依赖 ​

xml
<dependency>
    <groupId>org.apache.tika</groupId>
    <artifactId>tika-core</artifactId>
    <version>2.9.2</version>
</dependency>
<dependency>
    <groupId>org.apache.tika</groupId>
    <artifactId>tika-parsers-standard-package</artifactId>
    <version>2.9.2</version>
</dependency>

2，编写控制层代码逻辑 ​

java
public ResponseEntity<Object> readFile(@RequestParam("file") MultipartFile file) {
        if (file.isEmpty()) {
            return ResponseEntity.status(HttpStatus.BAD_REQUEST).body("上传文件不能为空");
        }

        try {
            // 获取文件名称
            String fileName = file.getOriginalFilename();

            // 使用 Tika 解析文件内容
            String content = tika.parseToString(file.getInputStream());

            return ResponseEntity.ok("文件名: " + fileName + "\n文件内容:\n" + content);

        } catch (IOException | TikaException e) {
            return ResponseEntity.status(HttpStatus.INTERNAL_SERVER_ERROR)
                    .body("文件解析失败: " + e.getMessage());
        }
    }

public ResponseEntity<Object> uploadMultipleFiles(@RequestParam("files") MultipartFile[] files) {
        if (files.length == 0) {
            return ResponseEntity.status(HttpStatus.BAD_REQUEST).body("上传文件不能为空");
        }

        List<String> fileDetails = new ArrayList<>();

        for (MultipartFile file : files) {
            if (file.isEmpty()) {
                fileDetails.add("文件为空，跳过处理！");
                continue;
            }

            try {
                // 获取文件名
                String fileName = file.getOriginalFilename();

                // 获取 MIME 类型（Spring 自带）
                String contentType = file.getContentType();

                // 使用 Apache Tika 自动识别文件类型
                String detectedType = tika.detect(file.getInputStream());

                // 获取文件扩展名
                String fileExtension = fileName.substring(fileName.lastIndexOf(".") + 1);

                fileDetails.add("文件名: " + fileName +
                        "\nMIME 类型: " + contentType +
                        "\nTika 识别类型: " + detectedType +
                        "\n文件扩展名: " + fileExtension +
                        "\n------------------------");

            } catch (IOException e) {
                fileDetails.add("文件解析失败: " + e.getMessage());
            }
        }

        return ResponseEntity.ok(fileDetails);
    }
