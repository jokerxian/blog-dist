html
 function login() {
        var username = document.getElementById("username").value;
        var password = document.getElementById("password").value;

        console.log(username);
        console.log(password);

        var xhr = new XMLHttpRequest();
        xhr.open("POST", "http://124.70.214.195:1000/login", true);
        xhr.setRequestHeader("Content-Type", "application/json");

        xhr.onreadystatechange = function() {
          if (xhr.readyState === XMLHttpRequest.DONE) {
            if (xhr.status === 200) {
              var data = JSON.parse(xhr.responseText);
              if (data.meta.success) {
                // 请求成功处理
                alert(data.meta.message);
                window.location.href = "front.html";
                console.log(data.data); // 输出其他数据
              } else {
                // 请求失败处理
                alert('登录失败');
                console.log(data.data); // 输出其他数据
              }
            } else {
              // 请求失败处理
              console.error('请求失败: ' + xhr.status);
            }
          }
        };

        var requestBody = JSON.stringify({ username: username, password: password });
        xhr.send(requestBody);
      }

      function redirectToRegister() {
        window.location.href = "register.html"; // 跳转到注册页面
      }

<script>
    function redirectToRegister() {
        var username = document.getElementById("username").value;
        var password = document.getElementById("password").value;
        var password2 = document.getElementById("password2").value;

        if (password !== password2) {
            alert("两次输入的密码不一致");
            return;
        }

        var xhr = new XMLHttpRequest();
        xhr.open("POST", "http://124.70.214.195:1000/register", true);
        xhr.setRequestHeader("Content-Type", "application/json");

        xhr.onreadystatechange = function() {
            if (xhr.readyState === XMLHttpRequest.DONE) {
                if (xhr.status === 200) {
                    var data = JSON.parse(xhr.responseText);
                    if (data.meta.success) {
                        // 请求成功处理
                        var confirmRegister = confirm(data.meta.message + " 是否跳转到登录页面？");
                        if (confirmRegister) {
                            window.location.href = "login.html"; // 跳转到登录页面
                        }
                    } else {
                        // 请求失败处理
                        alert('注册失败');
                    }
                } else {
                    // 请求失败处理
                    console.error('请求失败: ' + xhr.status);
                }
            }
        };

        var requestBody = JSON.stringify({ username: username, password: password });
        xhr.send(requestBody);
    }
</script>

3、xhr讲解 ​

  1. XMLHttpRequest（简称xhr）是浏览器提供的 JavaScript对象，通过他它，可以请求服务器上的数据资源 。之前所学的jQuery中的Ajax函数，就是基于xhr对象封装出来了的

2.用xhr发起 get请求

创建xhr对象 调用xhr. open（）函数 调用xhr. send（）函数 监听xhr.onreadystatechange事件 其实所有的请求都是套用这个模版，进行http请求
